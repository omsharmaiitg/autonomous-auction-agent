\section{Statistical Foundations}

\subsection{Skewness}
\begin{formulabox}[Formula: Skewness]
\[\text{Skewness}=\frac{\frac{1}{n}\sum_{i=1}^{n}(x_i-\bar{x})^3}{\sigma^3}\]
\end{formulabox}
\begin{itemize}
  \item $=0$: Symmetric $\rightarrow$ use \textbf{mean}
  \item $>0$: Right skew (long right tail, e.g.\ selling price) $\rightarrow$ use \textbf{median}
  \item $<0$: Left skew $\rightarrow$ use \textbf{median}
\end{itemize}

\subsection{Kurtosis}
\begin{formulabox}[Formula: Excess Kurtosis]
\[\text{Kurtosis}=\frac{\frac{1}{n}\sum_{i=1}^{n}(x_i-\bar{x})^4}{\sigma^4}-3\]
\end{formulabox}
$>0$: Heavy tails, more outliers. Our selling price had high kurtosis $\Rightarrow$ justified IQR removal and log-transform.

\subsection{IQR Outlier Removal}
\begin{formulabox}[IQR Rule]
\[\text{IQR}=Q_3-Q_1,\quad\text{bounds}=\bigl[Q_1-1.5\cdot\text{IQR},\;Q_3+1.5\cdot\text{IQR}\bigr]\]
\end{formulabox}

\section{Missing Values and Imputation}

\begin{definitionbox}[Imputation]
Replacing missing values with substitutes instead of deleting rows.
\end{definitionbox}

\subsection{Types of Missing Data}
\begin{enumerate}
  \item \textbf{MCAR} — Missing Completely At Random. Random failure. Safe to drop or impute.
  \item \textbf{MAR} — Missing At Random. Depends on other observed columns. Group-wise imputation.
  \item \textbf{MNAR} — Missing Not At Random. Depends on the missing value itself. Most dangerous.
\end{enumerate}

\subsection{Mean vs.\ Median}
\begin{center}
\begin{tabular}{lll}\toprule
  Distribution & Skewness & Use\\\midrule
  Symmetric & $|\text{skew}|<0.5$ & \textbf{Mean}\\
  Mildly skewed & $0.5$--$1.0$ & \textbf{Median}\\
  Highly skewed / outliers & $>1.0$ & \textbf{Median}\\\bottomrule
\end{tabular}
\end{center}

\section{Machine Learning Models}

\subsection{Decision Tree}
\begin{formulabox}[Splitting Criterion: MSE Reduction]
\[\text{Gain}=\text{MSE}(S)-\Bigl(\frac{|S_L|}{|S|}\text{MSE}(S_L)+\frac{|S_R|}{|S|}\text{MSE}(S_R)\Bigr)\]
\end{formulabox}
\textbf{Cons:} High variance, overfitting, greedy splits, unstable.

\subsection{Random Forest}
\begin{formulabox}[Prediction]
\[\hat{y}=\frac{1}{B}\sum_{b=1}^{B}T_b(\mathbf{x})\]
\end{formulabox}
Fixes decision tree via bootstrap aggregation (bagging) + random feature subsampling at each split.

\subsection{XGBoost}
Sequential boosting: each tree corrects residuals of previous trees.
\begin{formulabox}[Objective]
\[\mathcal{L}^{(t)}=\sum_i\ell\!\left(y_i,\hat{y}_i^{(t-1)}+f_t(\mathbf{x}_i)\right)+\gamma T+\tfrac{1}{2}\lambda\sum_j w_j^2\]
\end{formulabox}
Target transform: $y=\log_e(1+\text{price})$, invert: $\text{price}=e^{\hat{y}}-1$ (\texttt{np.expm1}).

\section{Regularisation, Bias-Variance, Loss Functions}

\subsection{Regularisation}
\begin{formulabox}[L1 (Lasso) and L2 (Ridge)]
\[\text{L1: }\mathcal{L}+\lambda\sum|w_i|\qquad\text{L2: }\mathcal{L}+\lambda\sum w_i^2\]
\end{formulabox}
L1 drives weights to exactly zero (feature selection). L2 shrinks all weights (handles multicollinearity).
\textbf{Geometric reason:} L1 diamond shape has corners on axes $\Rightarrow$ loss contours touch corners $\Rightarrow$ zero weights. L2 circle has no corners $\Rightarrow$ weights only shrink.

\subsection{Bias-Variance Tradeoff}
\[\text{Total Error}=\text{Bias}^2+\text{Variance}+\text{Irreducible Noise}\]
\begin{itemize}
  \item High Bias = underfitting (model too simple)
  \item High Variance = overfitting (model too complex)
\end{itemize}

\subsection{Loss Functions}
\begin{formulabox}[MSE (Linear Regression) and Cross-Entropy (Logistic)]
\[\text{MSE}=\frac{1}{n}\sum(y_i-\hat{y}_i)^2\]
\[\text{BCE}=-\frac{1}{n}\sum\bigl[y_i\log\hat{p}_i+(1-y_i)\log(1-\hat{p}_i)\bigr]\]
\end{formulabox}

\section{Neural Networks and CNN}

\subsection{Neural Network}
Each neuron: $z=\mathbf{w}^\top\mathbf{x}+b$, output $=f(z)$ (activation).
Training via backpropagation + gradient descent: $w\leftarrow w-\eta\frac{\partial\mathcal{L}}{\partial w}$.

\textbf{Vanishing gradient (Sigmoid problem):} $\sigma'(z)_{\max}=0.25$. Deep networks: $0.25^{10}\approx10^{-6}$. \textbf{Fix:} ReLU $=\max(0,z)$.

\subsection{CNN}
For images. Convolutional filters slide over image detecting local patterns (edges $\rightarrow$ shapes $\rightarrow$ objects). Shares weights $\Rightarrow$ far fewer parameters than dense NN.

\section{The Live Bidding Agent}

\subsection{Complete Bid Sequence Formula}
\begin{formulabox}[All Steps]
\begin{align*}
  V_{\text{adj}}&=\hat{p}\times0.95\\
  \sigma(r)&=\frac{1}{1+e^{-0.35(r-5)}}\\
  C(B)&=\begin{cases}0.88&B\geq0.6B_0\\0.70\text{--}0.80&0.3\leq B/B_0<0.6\\0.70&B<0.3B_0\end{cases}\\
  \text{max\_bid}&=\min\!\bigl(V_{\text{adj}}\cdot C(B)(1+0.1\sigma(r)),\;V_{\text{adj}},\;B\bigr)\\
  \text{next\_bid}&=\text{cur}+\max\!\bigl((0.40+0.50\,\sigma(r))\cdot\text{gap},\;\$50\bigr)
\end{align*}
If $\text{cur}\geq\text{max\_bid}$: \textbf{fold} (return 0.0).
\end{formulabox}

\subsection{Key Design Decisions}
\begin{itemize}
  \item \textbf{Hard cap at $V_{\text{adj}}$:} Profit guaranteed on every win
  \item \textbf{Proportional increment:} Conceals ceiling from rivals; closes faster in late rounds
  \item \textbf{Bankroll tiers:} Higher margin required as money depletes
  \item \textbf{Sigmoid:} Patient early, aggressive late — mimics expert bidder behaviour
  \item \textbf{Fold freely:} Win = positive profit only (confirmed rule); folding is never penalised
\end{itemize}

\[\text{RMSE}=\sqrt{\frac{1}{n}\sum(y_i-\hat{y}_i)^2}\qquad\text{MAPE}=\frac{1}{n}\sum\left|\frac{y_i-\hat{y}_i}{y_i}\right|\times100\%\]
\end{formulabox}
